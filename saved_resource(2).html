function getRecommendations() {
    const searchInput = document.getElementById('search').value;
    const resultsDiv = document.getElementById('results');

    // Sample data for recommendations (you can expand this)
    const recommendations = {
        beaches: ["Malibu Beach, California", "Bondi Beach, Australia"],
        temples: ["Golden Temple, India", "Temple of Heaven, China"],
        toronto: ["CN Tower", "Royal Ontario Museum"]
    };

    const matchedRecommendations = recommendations[searchInput.toLowerCase()] || [];

    resultsDiv.innerHTML = matchedRecommendations.length 
        ? `<h3>Recommendations:</h3><ul>${matchedRecommendations.map(item => `<li>${item}</li>`).join('')}</ul>`
        : "<p>No recommendations found.</p>";
}
